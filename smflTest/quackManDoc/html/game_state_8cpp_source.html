255 void PlayingState::update(sf::Time a_delta){
256  m_pacMan->update(a_delta);
257 
258  for(Ghost* ghost : m_ghosts){
259  ghost->update(a_delta);
260  }
261 
262  //check collision player->dots, player-ghosts, player->weak ghosts
263  //scores
264  // dot = 5
265  // super dot = 25
266  // weak ghost = 100
267  // bonus = 500
268 
269  sf::Vector2f pixelPosition = m_pacMan->getPosition();
270  sf::Vector2f offset(fmod(pixelPosition.x, 16), fmod(pixelPosition.y, 16));
271 
272  offset-= sf::Vector2f(8,8);
273 
274  if (offset.x <= 2 && offset.x >= -2 && offset.y <= 2 && offset.y >= -2){
275  sf::Vector2i cellPosition = m_maze.mapPixelToCell(pixelPosition);
276 
277  if (m_maze.isDot(cellPosition)){
278  m_score += 5;
279  }
280  else if (m_maze.isSuperDot(cellPosition)){
281  for (Ghost* ghost : m_ghosts){
282  ghost->setWeak(sf::seconds(5));
283  }
284 
285  m_score += 25;
286  }
287  m_maze.pickObject(cellPosition);
288  }
289 
290  for (Ghost* ghost : m_ghosts){
291  if (ghost->getCollision().intersects(m_pacMan->getCollision())){
292  if (ghost->isWeak()){
293  ghost->setWeak(sf::Time::Zero);
294  ghost->setPosition(m_maze.mapCellToPixel(m_maze.getRespawnPosition()));
295  m_score += 100;
296  }
297  else{
298  m_pacMan->die();
299  }
300  }
301  }
302 
303  if (m_pacMan->isDead()){
304  m_pacMan->reset();
305 
306  m_lives--;
307 
308  if (m_lives < 0)
309  getGame()->changeGameState(GameState::Lost);
310  else
311  resetCharacters();
312  }
313 
314  if(m_bonus.getCollision().intersects(m_pacMan->getCollision())){
315  m_bonus.setPosition(-16, -16);
316  m_score += 500;
317  }
318  if(m_maze.getRemainingDots() == 0){
319  getGame()->changeGameState(GameState::Won);
320  }
321  if(m_maze.getRemainingDots() == 150){
322  m_bonus.setPosition(m_maze.mapCellToPixel(m_maze.getRespawnPosition()));
323 
324  }
325  m_scoreText.setString(to_string(m_score));
326  m_dotsLeft.setString(to_string(m_maze.getRemainingDots()) + "x dots");
327 
328 }
